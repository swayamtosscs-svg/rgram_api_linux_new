# Environment Configuration for RGram API
# Copy this file to .env.local and fill in your actual values

# Database Configuration
MONGODB_URI=mongodb://localhost:27017/rgram_api
# For MongoDB Atlas, use: mongodb+srv://username:password@cluster.mongodb.net/database

# JWT Configuration
JWT_SECRET=your_super_secret_jwt_key_here_make_it_long_and_random
JWT_EXPIRE=7d

# Google OAuth Configuration
# Get these from Google Cloud Console: https://console.cloud.google.com/
GOOGLE_CLIENT_ID=your_google_client_id_here
GOOGLE_CLIENT_SECRET=your_google_client_secret_here
GOOGLE_CALLBACK_URL=http://localhost:3000/api/auth/google/callback

# CORS Configuration
CORS_ORIGIN=http://localhost:3000

# Google OAuth Mode
# Set to 'true' for testing without real Google OAuth (uses mock data)
# Set to 'false' for production with real Google OAuth
MOCK_GOOGLE_AUTH=true

# Node Environment
NODE_ENV=development

# Firebase Configuration (if using push notifications)
FIREBASE_PROJECT_ID=your_firebase_project_id
FIREBASE_PRIVATE_KEY=your_firebase_private_key
FIREBASE_CLIENT_EMAIL=your_firebase_client_email

# Email Configuration (if using email features)
EMAIL_HOST=smtp.gmail.com
EMAIL_PORT=587
EMAIL_USER=your_email@gmail.com
EMAIL_PASS=your_app_password
