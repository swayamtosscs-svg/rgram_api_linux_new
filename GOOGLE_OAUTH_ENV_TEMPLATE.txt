# Google OAuth Environment Variables Template
# Copy this file to .env.local and fill in your actual values

# Google OAuth Configuration
GOOGLE_CLIENT_ID=your_google_client_id_here
GOOGLE_CLIENT_SECRET=your_google_client_secret_here

# Next.js Configuration
NEXTAUTH_URL=https://api-rgram1.vercel.app
NEXTAUTH_SECRET=your_nextauth_secret_key_here

# JWT Configuration (for production, use a strong secret)
JWT_SECRET=your_jwt_secret_key_here

# Database Configuration (if using MongoDB)
MONGODB_URI=mongodb://localhost:27017/api_rgram1
# OR for MongoDB Atlas:
# MONGODB_URI=mongodb+srv://username:password@cluster.mongodb.net/database_name

# Environment
NODE_ENV=development

# Instructions:
# 1. Go to Google Cloud Console: https://console.cloud.google.com/
# 2. Create a new project or select existing one
# 3. Enable Google+ API and Google OAuth2 API
# 4. Go to Credentials > Create Credentials > OAuth 2.0 Client IDs
# 5. Set Application Type to "Web application"
# 6. Add authorized redirect URIs:
#    - https://api-rgram1.vercel.app/api/auth/google/callback
#    - http://localhost:3000/api/auth/google/callback (for development)
# 7. Copy Client ID and Client Secret
# 
# Note: The callback will redirect to /google-oauth-demo.html instead of /auth/social-callback
